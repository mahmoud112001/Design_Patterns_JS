QUESTION 01
CEO — Singleton Pattern
A company has only ONE CEO. No matter how many times you try to create a new CEO, you always get the same instance back. Try it below.
Singleton
// CEO.js
class CEO {
  constructor(name, age, address) {
    if (CEO.instance) {
      return CEO.instance;
    }
    this.name    = name;
    this.age     = age;
    this.address = address;
    CEO.instance = this;
  }
}

// Only the FIRST call creates an instance
const ceo1 = new CEO("Mahmoud", 25, "Alexandria");
const ceo2 = new CEO("Ahmed", 30, "Cairo");

console.log(ceo1 === ceo2); // true

Guarantees one global instance — perfect for CEO, config, DB connection
QUESTION 02
Vehicle Shop — Factory Pattern
A car shop handles Cars, Trucks, and Motorcycles daily. The Factory creates the right vehicle object without exposing the constructor logic. Select a type and build a vehicle.
Factory
// VehicleFactory.js
class VehicleFactory {
  createVehicle(type, brand, model) {
    switch (type.toLowerCase()) {
      case "car":
        return new Car(type, brand, model);
      case "truck":
        return new Truck(type, brand, model);
      case "motorcycle":
        return new Motorcycle(type, brand, model);
    }
  }
}

const factory = new VehicleFactory();
const v1 = factory.createVehicle(
  "car", "Toyota", "Corolla"
);

QUESTION 03
Toy Factory — Factory Pattern
ToyFactory creates ToyDuck (color, price) or ToyCar (color, price, name) using prototype-based constructors. Select a toy type and fill in its properties.
Factory
// ToyFactory.js
function ToyDuck(color, price) {
  this.color = color;
  this.price = price;
}

function ToyCar(color, price, name) {
  this.color = color;
  this.price = price;
  this.name  = name;
}

ToyFactory.prototype.createToy =
  function(type, color, price, name) {
  switch (type.toLowerCase()) {
    case "duck":
      return new ToyDuck(color, price);
    case "car":
      return new ToyCar(color, price, name);
    default:
      throw new Error("Invalid type");
  }
};

QUESTION 04
ConfigureVals — Singleton (IIFE Closure)
getConfiguration() uses an IIFE closure to ensure only one ConfigureVals instance ever exists. Second call always returns the first instance. Try creating two configs below.
Singleton · IIFE
// ConfigureVals.js
class ConfigureVals {
  constructor(
    xpoint = 0,
    ypoint = 0,
    shape  = null
  ) {
    this.xpoint = xpoint;
    this.ypoint = ypoint;
    this.shape  = shape;
  }
}

const getConfiguration = (function () {
  let instance = null;
  return function(xpoint, ypoint, shape) {
    if (!instance) {
      instance = new ConfigureVals(
        xpoint, ypoint, shape
      );
    }
    return instance;
  };
})(); // ← IIFE executes immediately

IIFE CLOSURE

VS Q1 SINGLETON
